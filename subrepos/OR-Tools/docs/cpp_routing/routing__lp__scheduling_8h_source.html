26 // Utility class used in the core optimizer to tighten the cumul bounds as much
27 // as possible based on the model precedences.
28 class CumulBoundsPropagator {
29  public:
30  explicit CumulBoundsPropagator(const RoutingDimension* dimension);
31 
32  // Tightens the cumul bounds starting from the current cumul var min/max,
33  // and propagating the precedences resulting from the next_accessor, and the
34  // dimension's precedence rules.
35  // Returns false iff the precedences are infeasible with the given routes.
36  // Otherwise, the user can call CumulMin() and CumulMax() to retrieve the new
37  // bounds of an index.
38  bool PropagateCumulBounds(const std::function<int64(int64)>& next_accessor,
39  int64 cumul_offset);
40 
41  int64 CumulMin(int index) const {
42  return propagated_bounds_[PositiveNode(index)];
43  }
44 
45  int64 CumulMax(int index) const {
46  const int64 negated_upper_bound = propagated_bounds_[NegativeNode(index)];
47  return negated_upper_bound == kint64min ? kint64max : -negated_upper_bound;
48  }
49 
50  private:
51  // An arc "tail --offset--> head" represents the relation
52  // tail + offset <= head.
53  // As arcs are stored by tail, we don't store it in the struct.
54  struct ArcInfo {
55  int head;
56  int64 offset;
57  };
58  static const int kNoParent;
59  static const int kParentToBePropagated;
60 
61  // Return the node corresponding to the lower bound of the cumul of index and
62  // -index respectively.
63  int PositiveNode(int index) const { return 2 * index; }
64  int NegativeNode(int index) const { return 2 * index + 1; }
65 
66  void AddNodeToQueue(int node) {
67  if (!node_in_queue_[node]) {
68  bf_queue_.push_back(node);
69  node_in_queue_[node] = true;
70  }
71  }
72 
73  // Adds the relation first_index + offset <= second_index, by adding arcs
74  // first_index --offset--> second_index and
75  // -second_index --offset--> -first_index.
76  void AddArcs(int first_index, int second_index, int64 offset);
77 
78  bool InitializeArcsAndBounds(const std::function<int64(int64)>& next_accessor,
79  int64 cumul_offset);
80 
81  bool UpdateCurrentLowerBoundOfNode(int node, int64 new_lb);
82 
83  bool DisassembleSubtree(int source, int target);
84 
85  bool CleanupAndReturnFalse() {
86  // We clean-up node_in_queue_ for future calls, and return false.
87  for (int node_to_cleanup : bf_queue_) {
88  node_in_queue_[node_to_cleanup] = false;
89  }
90  bf_queue_.clear();
91  return false;
92  }
93 
94  const RoutingDimension& dimension_;
95  const int64 num_nodes_;
96 
97  // TODO(user): Investigate if all arcs for a given tail can be created
98  // at the same time, in which case outgoing_arcs_ could point to an absl::Span
99  // for each tail index.
100  std::vector<std::vector<ArcInfo>> outgoing_arcs_;
101 
102  std::deque<int> bf_queue_;
103  std::vector<bool> node_in_queue_;
104  std::vector<int> tree_parent_node_of_;
105  // After calling PropagateCumulBounds(), for each node index n,
106  // propagated_bounds_[2*n] and -propagated_bounds_[2*n+1] respectively contain
107  // the propagated lower and upper bounds of n's cumul variable.
108  std::vector<int64> propagated_bounds_;
109 
110  // Vector used in DisassembleSubtree() to avoid memory reallocation.
111  std::vector<int> tmp_dfs_stack_;
112 };

218 // Class used to compute optimal values for dimension cumuls of routes,
219 // minimizing cumul soft lower and upper bound costs, and vehicle span costs of
220 // a route.
221 // In its methods, next_accessor is a callback returning the next node of a
222 // given node on a route.
223 class LocalDimensionCumulOptimizer {
224  public:
225  explicit LocalDimensionCumulOptimizer(const RoutingDimension* dimension);
226 
227  // If feasible, computes the optimal cost of the route performed by a vehicle,
228  // minimizing cumul soft lower and upper bound costs and vehicle span costs,
229  // and stores it in "optimal_cost" (if not null).
230  // Returns true iff the route respects all constraints.
231  bool ComputeRouteCumulCost(int vehicle,
232  const std::function<int64(int64)>& next_accessor,
233  int64* optimal_cost);
234 
235  // Same as ComputeRouteCumulCost, but the cost computed does not contain
236  // the part of the vehicle span cost due to fixed transits.
237  bool ComputeRouteCumulCostWithoutFixedTransits(
238  int vehicle, const std::function<int64(int64)>& next_accessor,
239  int64* optimal_cost_without_transits);
240 
241  // If feasible, computes the optimal cumul values of the route performed by a
242  // vehicle, minimizing cumul soft lower and upper bound costs and vehicle span
243  // costs, stores them in "optimal_cumuls" (if not null), and returns true.
244  // Returns false if the route is not feasible.
245  bool ComputeRouteCumuls(int vehicle,
246  const std::function<int64(int64)>& next_accessor,
247  std::vector<int64>* optimal_cumuls);
248 
249  // Similar to ComputeRouteCumuls, but also tries to pack the cumul values on
250  // the route, such that the cost remains the same, the cumul of route end is
251  // minimized, and then the cumul of the start of the route is maximized.
252  bool ComputePackedRouteCumuls(
253  int vehicle, const std::function<int64(int64)>& next_accessor,
254  std::vector<int64>* packed_cumuls);
255 
256  const RoutingDimension* dimension() const {
257  return optimizer_core_.dimension();
258  }
259 
260  private:
261  std::vector<std::unique_ptr<glop::LPSolver>> lp_solver_;
262  std::vector<std::unique_ptr<glop::LinearProgram>> linear_program_;
263  DimensionCumulOptimizerCore optimizer_core_;
264 };
