472 template <bool swap_first>
473 bool PairNodeSwapActiveOperator<swap_first>::MakeNextNeighbor(
474  Assignment* delta, Assignment* deltadelta) {
475  while (inactive_pair_ < pairs_.size()) {
476  if (!IsInactive(pairs_[inactive_pair_].first[0]) ||
477  !IsInactive(pairs_[inactive_pair_].second[0]) ||
478  !PathOperator::MakeNextNeighbor(delta, deltadelta)) {
479  ResetPosition();
480  ++inactive_pair_;
481  } else {
482  return true;
483  }
484  }
485  return false;
486 }

488 template <bool swap_first>
489 bool PairNodeSwapActiveOperator<swap_first>::MakeNeighbor() {
490  const int64 base = BaseNode(0);
491  if (IsPathEnd(base)) {
492  return false;
493  }
494  const int64 pair_first = pairs_[inactive_pair_].first[0];
495  const int64 pair_second = pairs_[inactive_pair_].second[0];
496  if (swap_first) {
497  return MakeActive(pair_second, BaseNode(1)) &&
498  MakeActive(pair_first, base) &&
499  MakeChainInactive(pair_first, Next(pair_first));
500  } else {
501  return MakeActive(pair_second, BaseNode(1)) &&
502  MakeActive(pair_first, base) &&
503  MakeChainInactive(pair_second, Next(pair_second));
504  }
505 }
